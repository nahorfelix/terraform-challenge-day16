4) LIFECYCLE
prevent_destroy on state bucket + lock table: stops accidental destroy of remote state — without it, deleted state = lost infra map and possible prolonged outage.
create_before_destroy on LT/SG/ASG: new resource before delete — without in-place updates, destroys first can cause downtime or dependency errors.

5) CLOUDWATCH ALARM (paste)
resource aws_cloudwatch_metric_alarm high_cpu {
  count = local.monitoring_enabled ? 1 : 0
  alarm_name = "${var.cluster_name}-high-cpu"
  comparison_operator = "GreaterThanThreshold"
  evaluation_periods = 2
  metric_name = "CPUUtilization"
  namespace = "AWS/EC2"
  period = 120
  statistic = "Average"
  threshold = var.cpu_alarm_threshold
  dimensions = { AutoScalingGroupName = aws_autoscaling_group.web.name }
  alarm_actions = concat([aws_sns_topic.alerts.arn], local.autoscaling_enabled ? [aws_autoscaling_policy.scale_out[0].arn] : [])
}
Threshold default 80%: two 2-minute periods above 80% = ~4 min sustained load before alarm; avoids flapping. When fired: SNS delivers; optional autoscaling policy adds capacity.

func TestWebserverCluster(t *testing.T) {
  t.Parallel()
  terraformOptions := terraform.WithDefaultRetryableErrors(t, &terraform.Options{

      "enable_detailed_monitoring":false,

  defer terraform.Destroy(t, terraformOptions)
  terraform.InitAndApply(t, terraformOptions)
  albDns := terraform.Output(t, terraformOptions, "alb_dns_name")
  http_helper.HttpGetWithRetry(t, "http://"+albDns, nil, 200, "Hello World", 30, 10*time.Second)
}
